flowchart TD
    START["👤 Recruiter submits intake form\n5-step wizard with AI pre-fill"]
    START --> DB1[("💾 intake_requests\nstatus: draft")]
    DB1 --> JOB["compute_job created\ntype: generate"]
    JOB --> WORKER["🔧 Python Worker\nclaims job atomically via SELECT FOR UPDATE"]

    %% STAGE 1
    WORKER --> S1["━━ Stage 1: Strategic Intelligence ━━"]
    S1 --> S1a["Kimi K2.5 — analyze RFP/intake data"]
    S1a --> S1b["Generate:\n• Creative brief\n• Dynamic personas (2-5)\n• Cultural research per region\n• Design direction\n• Derived requirements"]
    S1b --> S1c["Campaign strategy\nbudgets, channels, ad sets, targeting"]
    S1c --> DB2[("💾 creative_briefs\n+ campaign_strategies")]

    %% STAGE 2
    DB2 --> S2["━━ Stage 2: Character-Driven Images ━━"]
    S2 --> S2a["Create actor profiles\n1 per persona, scene-aware prompts"]
    S2a --> S2b["Seedream 4.5\ngenerate 4 scenes per actor"]
    S2b --> S2c{"🛡️ VQA Gate\nface quality · artifacts\nhex colors · dirty rooms"}
    S2c -->|"PASS"| S2d["Upload to Vercel Blob\nbase_image assets"]
    S2c -->|"FAIL"| S2e["Retry with adjusted prompt\nup to 3 attempts"]
    S2e --> S2b
    S2d --> DB3[("💾 actor_profiles\n+ generated_assets: base_image")]

    %% STAGE 3
    DB3 --> S3["━━ Stage 3: Copy Generation ━━"]
    S3 --> S3a["Gemma 4 — layered context\n2K base knowledge + 1K project context"]
    S3a --> S3b["Per persona × channel × language\n3 pillar variations each"]
    S3b --> S3c["Output: headlines, hooks,\nbody, CTAs, descriptions"]
    S3c --> S3d{"🛡️ Copy Quality Gate\nbrand voice · accuracy · tone"}
    S3d -->|"PASS"| DB4[("💾 generated_assets: copy")]
    S3d -->|"FAIL"| S3e["Retry with feedback"]
    S3e --> S3a

    %% STAGE 4
    DB4 --> S4["━━ Stage 4: Layout Composition ━━"]
    S4 --> S4a["GLM-5 designs HTML compositions\nusing reference templates"]
    S4a --> S4b["Scene-aware actor placement\npillar-matched scene selection"]
    S4b --> S4c["Graphic copy overlay\nper-language, 25% budget"]
    S4c --> S4d["Playwright render → PNG\nheadless Chrome screenshot"]
    S4d --> S4e{"🛡️ Creative VQA Gate\nperson visible · text legible\nno artifacts · score ≥ 0.85"}
    S4e -->|"PASS 85%+"| S4f["Upload composed creatives\nto Vercel Blob"]
    S4e -->|"FAIL"| S4g["Flux 2 edit loop\ninpaint problems, up to 3 iterations"]
    S4g --> S4d
    S4f --> DB5[("💾 generated_assets:\ncomposed_creative + carousel_panel")]

    %% STAGE 5
    DB5 --> S5["━━ Stage 5: Video Generation ━━"]
    S5 --> S5a["Kling 3.0 — 12-15s vertical\nUGC-style talking head"]
    S5a --> S5b["Multi-shot with native\nlip sync + voice"]
    S5b --> DB6[("💾 generated_assets: video")]

    %% STAGE 6 — NEW
    DB6 --> S6["━━ 🟠 Stage 6: Landing Pages ━━"]
    S6 --> S6a["Select template variant\nbased on task_type + persona"]
    S6a --> S6b["Extract hard facts\ncompensation · quals · locations · URLs\nINJECTED, never LLM-written"]
    S6b --> S6c["Pull best Stage 3 copy\nhero H1 from best headline\nCTA from best call-to-action"]
    S6c --> S6d["Gemma 4 — fresh sections\nwhy cards · activities · sessions · FAQ"]
    S6d --> S6e["Jinja2 render\nfull self-contained HTML\nwith images from Stage 2+4"]
    S6e --> S6f{"🛡️ Drift Validation Gate\ncompensation match · qual match\nCTA URL match · work_mode check\nlanguage check"}
    S6f -->|"ALL MATCH"| S6g["Upload HTML to Vercel Blob"]
    S6f -->|"MISMATCH"| S6h["❌ Log error\nmark evaluation_passed: false"]
    S6g --> DB7[("💾 generated_assets: landing_page\n+ /lp/slug route ready")]

    %% PIPELINE COMPLETE
    DB7 --> DONE["✅ All 6 stages complete"]
    DONE --> STATUS[("💾 intake_requests\nstatus → review")]
    STATUS --> N1["📣 Teams Notification\n'Creatives ready for review'"]

    %% 3-STAGE APPROVAL
    N1 --> MKT{"🚦 Marketing Manager Review\nedits strategy, reviews creatives"}
    MKT -->|"Approve"| DWORK["🎨 Designer works\nedits · retouches · Figma"]
    MKT -->|"Reject"| REGEN["🔄 Trigger regeneration\nfull or per-stage"]
    REGEN --> WORKER

    DWORK --> DSUB["Designer submits finals"]
    DSUB --> DGATE{"🚦 Designer Approval\nrecorded in approvals table"}

    DGATE -->|"Approved"| FGATE{"🚦 Final Approval\nMarketing Manager sign-off"}
    FGATE -->|"Approve"| SENT[("💾 status → sent")]

    SENT --> MAGIC["🔗 Generate agency magic link\ntoken-authenticated URL"]
    MAGIC --> N2["📣 Teams Notification\nmagic link sent to agency"]
    MAGIC --> AGENCY["Agency opens portal\n/agency/id?token=..."]

    %% LANDING PAGE SERVING
    DB7 --> SERVE["🌐 /lp/slug serves pages\npublic, no auth required"]
